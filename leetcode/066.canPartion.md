# 66.分割等和子集

def canPartition(self, nums):
    total = sum(nums)
    if total % 2 == 1:  # 总和是奇数，肯定分不了
        return False
    
    target = total // 2
    n = len(nums)
    
    # dp[i][j] 表示前 i 个数能不能凑出和为 j
    dp = [[False] * (target + 1) for _ in range(n + 1)]
    
    # 凑出和为 0 总是可以的（什么都不选）
    for i in range(n + 1):
        dp[i][0] = True
    
    for i in range(1, n + 1):
        for j in range(1, target + 1):
            # 不选第 i 个数
            dp[i][j] = dp[i-1][j]
            # 选第 i 个数（如果能装下）
            if j >= nums[i-1]:
                dp[i][j] = dp[i][j] or dp[i-1][j-nums[i-1]]
    
    return dp[n][target]
