def rob(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        # 定义一个递归函数
        # 返回值是一个数组：[不偷当前节点的最大金额, 偷当前节点的最大金额]
        def dfs(node):
            # 1. 终止条件：如果是空节点，偷和不偷都是 0
            if not node:
                return [0, 0]
            
            # 2. 递归计算左右孩子（后序遍历：左 -> 右 -> 根）
            left = dfs(node.left)   # left[0]是不偷左孩子，left[1]是偷左孩子
            right = dfs(node.right) # right[0]是不偷右孩子，right[1]是偷右孩子
            
            # 3. 计算当前节点的状态
            # 情况1：不偷当前节点
            # 孩子可以偷，也可以不偷，我们要取最大值
            val_not_rob = max(left[0], left[1]) + max(right[0], right[1])
            
            # 情况2：偷当前节点
            # 孩子绝对不能偷（只能取孩子不偷的状态）+ 当前节点的钱
            val_rob = node.val + left[0] + right[0]
            
            return [val_not_rob, val_rob]
        
        # 从根节点开始递归
        result = dfs(root)
        
        # 返回根节点偷或不偷的最大值
        return max(result[0], result[1])
