# 63.前k个高频元素

```python
def topKFrequent(self, nums, k):
    """
    :type nums: List[int]
    :type k: int
    :rtype: List[int]
    """
    from collections import defaultdict
    
    # 1. 统计每个数字出现的次数
    di = defaultdict(int)  # 这里要加 int，表示默认值是 0
    for num in nums:  
        di[num] += 1
    
    # 2. 按出现次数排序，取前 k 个
    # di.items() 返回的是 (数字, 次数) 的元组
    # key=lambda x: x[1] 表示按次数（元组的第二个元素）排序
    # sorted() 返回排序后的列表，reverse=True 表示降序（次数多的在前）
    sorted_items = sorted(di.items(), key=lambda x: x[1], reverse=True)
    
    # 3. 提取前 k 个数字
    result = [item[0] for item in sorted_items[:k]]
    
    return result

```
时间复杂度 O(nlogn)

```python
def topKFrequent(self, nums, k):
    """
    :type nums: List[int]
    :type k: int
    :rtype: List[int]
    """
    # 1. 统计频率 O(n)
    count = {}
    for num in nums:
        count[num] = count.get(num, 0) + 1
    
    # 2. 桶排序 O(n)
    # 桶的下标是频率（1 到 n），所以长度是 n+1
    buckets = [[] for _ in range(len(nums) + 1)]
    
    for num, freq in count.items():
        buckets[freq].append(num)
        
    # 3. 从后往前遍历桶，收集前 k 个 O(n)
    res = []
    # 从频率最高的桶（len(nums)）开始倒着往前找
    for i in range(len(buckets) - 1, 0, -1):
        if buckets[i]:
            res.extend(buckets[i])
        if len(res) >= k:
            break
            
    return res[:k]

```
时间复杂度O(n)
